## 3. 꼬일 수 있는 위험 요소

### 3.1 iOS 버전 분기가 이미 BaseSelectMode에 있음

```swift
// enterSelectMode(), exitSelectMode()에서 자동 분기
if #available(iOS 26.0, *) {
    enterSelectModeSystemUI()  // → 공통
} else {
    enterSelectModeFloatingUI()  // → Trash만 오버라이드
}
```

→ 이 분기를 건드리면 모든 VC에 영향

### 3.2 restoreNavigationBarAfterSelectMode() 기본 구현

```swift
// BaseSelectMode.swift:49-58
@objc func restoreNavigationBarAfterSelectMode() {
    if #available(iOS 26.0, *) {
        navigationItem.rightBarButtonItem = UIBarButtonItem(...)  // 단수!
    }
}
```

→ Grid는 이걸 오버라이드 안 해서 기본 구현 사용 중
→ 그런데 Grid 초기 설정에서는 `rightBarButtonItems` 복수 사용
→ 초기 설정과 복원 로직이 불일치할 수 있음

### 3.3 FloatingUI exitSelectModeFloatingUI() 기본 구현

```swift
// BaseSelectMode.swift:260-268
@objc func exitSelectModeFloatingUI() {
    overlay.titleBar.exitSelectMode()
    overlay.tabBar.exitSelectMode(animated: true)
    // 버튼 복원 없음!
}
```

→ Grid는 이걸 오버라이드 안 함
→ `configureFloatingOverlay()`가 viewWillAppear에서 호출되어 복원되는 것으로 추정

## 4. 안전한 리팩토링 전략

### 접근 방식: 2단계로 분리

**Phase A: 최소 헬퍼 도입 (안전)**

## 5. Phase A: 최소 헬퍼 도입

### 5.1 새 파일: NavigationButtonHelpers.swift

```swift
// Features/Grid/NavigationButtonHelpers.swift

import UIKit

extension BaseGridViewController {

    // MARK: - 버튼 생성 헬퍼

    /// Select 버튼 생성
    @available(iOS 26.0, *)
    func createSelectBarButtonItem() -> UIBarButtonItem {
        UIBarButtonItem(
            title: "Select",
            style: .plain,
            target: self,
            action: #selector(selectButtonTapped)
        )
    }

    /// 정리 버튼 생성 (Grid/Album용)
    @available(iOS 26.0, *)
    func createCleanupBarButtonItem(action: Selector) -> UIBarButtonItem {
        UIBarButtonItem(
            title: "정리",
            style: .plain,
            target: self,
            action: action
        )
    }

    /// 비우기 버튼 생성 (Trash용)
    @available(iOS 26.0, *)
    func createEmptyTrashBarButtonItem(action: Selector) -> UIBarButtonItem {
        let item = UIBarButtonItem(
            title: "비우기",
            style: .plain,
            target: self,
            action: action
        )
        item.tintColor = .systemRed
        return item
    }

    // MARK: - FloatingUI 버튼 설정 헬퍼

    /// Select + 정리 버튼 설정 (Grid/Album용)
    func setupFloatingSelectAndCleanupButtons(
        overlay: FloatingOverlayContainer,
        cleanupAction: @escaping () -> Void,
        isEmpty: Bool
    ) {
        overlay.titleBar.setTwoRightButtons(
            firstTitle: "Select",
            firstColor: .systemBlue,
            firstAction: { [weak self] in self?.selectButtonTapped() },
            secondTitle: "정리",
            secondColor: .systemBlue,
            secondAction: cleanupAction
        )
        overlay.titleBar.setTwoRightButtonsEnabled(
            firstEnabled: !isEmpty,
            secondEnabled: !isEmpty
        )
    }

    /// Select + 비우기 버튼 설정 (Trash용)
    func setupFloatingSelectAndEmptyButtons(
        overlay: FloatingOverlayContainer,
        emptyAction: @escaping () -> Void,
        isEmpty: Bool
    ) {
        overlay.titleBar.setTwoRightButtons(
            firstTitle: "Select",
            firstColor: .systemBlue,
            firstAction: { [weak self] in self?.enterSelectMode() },
            secondTitle: "비우기",
            secondColor: .systemRed,
            secondAction: emptyAction
        )
        overlay.titleBar.setTwoRightButtonsEnabled(
            firstEnabled: !isEmpty,
            secondEnabled: !isEmpty
        )
    }
}
```

### 5.2 AlbumGridViewController 헬퍼 적용

#### 헬퍼 적용 범위

헬퍼는 **버튼 설정만** 담당하며, 타이틀/뒤로가기 설정은 기존 코드 유지:

| 기존 코드 | 처리 방식 |
|----------|----------|
| `setTitle(albumTitle)` | **유지** (헬퍼 범위 밖) |
| `setShowsBackButton(true)` | **유지** (헬퍼 범위 밖) |
| `resetToSelectButton()` | **교체** → 헬퍼 호출 |
| `isSelectButtonHidden/Enabled` | **제거** (헬퍼가 내부 처리) |

#### iOS 26+ 수정

**수정 전** (AlbumGridViewController.swift:235-253):
```swift
@available(iOS 26.0, *)
private func setupSystemNavigationBarForAlbum() {
    let selectButton = UIBarButtonItem(...)
    navigationItem.rightBarButtonItem = selectButton  // 단수 - Select만
}
```

**수정 후**:
```swift
@available(iOS 26.0, *)
private func setupSystemNavigationBarForAlbum() {
    let selectButton = createSelectBarButtonItem()
    let cleanupButton = createCleanupBarButtonItem(action: #selector(cleanupButtonTapped))

    let isEmpty = gridDataSource.assetCount == 0
    selectButton.isEnabled = !isEmpty
    cleanupButton.isEnabled = !isEmpty

    navigationItem.rightBarButtonItems = [selectButton, cleanupButton]  // 복수
}
```

#### iOS 16~25 수정

**수정 전** (AlbumGridViewController.swift:208-231):
```swift
private func configureFloatingOverlayForAlbum() {
    overlay.titleBar.setTitle(albumTitle)
    overlay.titleBar.setShowsBackButton(true) { ... }
    overlay.titleBar.resetToSelectButton()           // ← 헬퍼로 교체
    overlay.titleBar.isSelectButtonHidden = false
    overlay.titleBar.isSelectButtonEnabled = !isEmpty
}
```

**수정 후**:
```swift
private func configureFloatingOverlayForAlbum() {
    guard let overlay = ... else { return }

    // ── 기존 유지 (헬퍼 범위 밖) ──
    overlay.titleBar.setTitle(albumTitle)
    overlay.titleBar.setShowsBackButton(true) { [weak self] in
        self?.navigationController?.popViewController(animated: true)
    }

    // ── 버튼 설정 (헬퍼로 교체) ──
    let isEmpty = gridDataSource.assetCount == 0
    setupFloatingSelectAndCleanupButtons(
        overlay: overlay,
        cleanupAction: { [weak self] in self?.cleanupButtonTapped() },
        isEmpty: isEmpty
    )
    // 기존 resetToSelectButton(), isSelectButtonHidden, isSelectButtonEnabled 제거
}
```

### 5.3 정리: GridViewController Select 복원 추가

GridViewController는 현재 `restoreNavigationBarAfterSelectMode()`를 오버라이드하지 않아서 Base 기본 구현(Select만) 사용 중.

#### 주의사항

1. **isEnabled 복원 필수**: 빈 상태에서 Select 종료 시 버튼이 활성화되지 않도록 초기 설정과 동일한 비활성 로직 필요
2. **configureFloatingOverlay 재사용 권장**: exitSelectModeFloatingUI()에서 직접 버튼 설정 시 타이틀/레이아웃 등 다른 UI 상태 누락 위험

**GridSelectMode.swift에 추가**:
```swift
// GridViewController extension

// iOS 26+: 버튼 + 활성화 상태 복원
override func restoreNavigationBarAfterSelectMode() {
    if #available(iOS 26.0, *) {
        let selectButton = createSelectBarButtonItem()
        let cleanupButton = createCleanupBarButtonItem(action: #selector(showCleanupEntry))

        // ⚠️ 빈 상태면 버튼 비활성화 (초기 설정과 동일)
        let isEmpty = gridDataSource.assetCount == 0
        selectButton.isEnabled = !isEmpty
        cleanupButton.isEnabled = !isEmpty

        navigationItem.rightBarButtonItems = [selectButton, cleanupButton]
    }
}

// iOS 16~25: configureFloatingOverlay() 재사용으로 일관성 유지
override func exitSelectModeFloatingUI() {
    guard let tabBarController = tabBarController as? TabBarController,
          let overlay = tabBarController.floatingOverlay else { return }

    overlay.titleBar.exitSelectMode()
    overlay.tabBar.exitSelectMode(animated: true)

    // 기존 configureFloatingOverlay() 재사용 (타이틀/버튼/활성화 상태 일괄 처리)
    configureFloatingOverlay()
}
```

protocol NavigationButtonConfigurable {

    var buttonsEnabled: Bool { get }

struct NormalModeButtonConfig {
    let selectButton: ButtonInfo
    let secondButton: ButtonInfo?
}

struct ButtonInfo {
    let title: String
    let tintColor: UIColor
    let action: () -> Void
}

## 11. 주의사항

1. **Trash의 특수 로직은 변경 금지**
   - `enterTrashSelectMode()`, `exitTrashSelectMode()` 호출 유지
   - `FloatingTabBarDelegate` 구현 유지

2. **Album의 뒤로가기 버튼은 별도 설정**
   - `setShowsBackButton()`은 버튼 헬퍼와 분리

3. **iOS 버전 분기는 각 호출부에서 유지**
   - `if #available(iOS 26.0, *)` 분기는 유지
   - 헬퍼는 iOS 26+ 전용으로 `@available` 표시

4. **테스트 충분히 수행**
   - Phase A Step별로 테스트
   - 전체 완료 후 회귀 테스트
