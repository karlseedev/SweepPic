## 데이터 소스 추상화 설계

### 문제점
세 파일이 각각 다른 방식으로 데이터에 접근:
```swift
// GridViewController
dataSourceDriver.count
dataSourceDriver.asset(at: indexPath)
dataSourceDriver.assetID(at: indexPath)
dataSourceDriver.indexPath(for: assetID)

// AlbumGridViewController
fetchResult.count
fetchResult.object(at: index)
fetchResult.object(at: index).localIdentifier
// 수동 for 루프로 검색

// TrashAlbumViewController
trashedAssets.count
trashedAssets[index]
trashedAssets[index].localIdentifier
trashedAssets.firstIndex(where:)
```

### 해결책: GridDataSource 프로토콜

```swift
/// 그리드 데이터 소스 프로토콜
/// BaseGridViewController에서 데이터 접근을 추상화
protocol GridDataSource: AnyObject {
    /// 전체 에셋 개수
    var assetCount: Int { get }

    /// 인덱스에 해당하는 PHAsset 반환
    func asset(at index: Int) -> PHAsset?

    /// 인덱스에 해당하는 에셋 ID 반환
    func assetID(at index: Int) -> String?

    /// 에셋 ID로 인덱스 검색
    func assetIndex(for assetID: String) -> Int?

    /// 뷰어용 PHFetchResult (옵셔널)
    var fetchResultForViewer: PHFetchResult<PHAsset>? { get }

    /// fetchResultForViewer가 nil인 경우 사용할 에셋 ID 배열
    /// 뷰어에서 순서 보장된 네비게이션에 사용
    var orderedAssetIDs: [String] { get }
}
```

### 구현체

### 성능 최적화 전략

**assetIndex(for:) O(N) 문제 해결:**
- `PHFetchResult`는 내부적으로 `index(of:)` 메서드가 최적화되어 있음
- `[PHAsset]` 배열(Trash)은 O(N) 선형 탐색 발생 가능
- 해결책: `TrashDataSource`에 `[String: Int]` 인덱스 캐시 추가

```swift
final class TrashDataSource: GridDataSource {
    // ⚠️ private: 외부에서 직접 변경 시 indexCache가 stale됨
    private(set) var assets: [PHAsset] = []

    // 인덱스 캐시: assetID -> index
    private var indexCache: [String: Int] = [:]

    /// 에셋 배열 갱신 시 캐시도 함께 갱신 (유일한 변경 경로)
    func updateAssets(_ newAssets: [PHAsset]) {
        assets = newAssets
        rebuildIndexCache()
    }

    private func rebuildIndexCache() {
        indexCache = Dictionary(uniqueKeysWithValues:
            assets.enumerated().map { ($1.localIdentifier, $0) }
        )
    }

    func assetIndex(for assetID: String) -> Int? {
        return indexCache[assetID]  // O(1)
    }

    // GridDataSource 구현
    var assetCount: Int { assets.count }
    func asset(at index: Int) -> PHAsset? {
        guard index >= 0, index < assets.count else { return nil }
        return assets[index]
    }
    func assetID(at index: Int) -> String? {
        return asset(at: index)?.localIdentifier
    }
    var fetchResultForViewer: PHFetchResult<PHAsset>? { nil }
    var orderedAssetIDs: [String] { assets.map { $0.localIdentifier } }
}
```

**fetchResultForViewer nil 문제 해결:**
- `TrashDataSource`는 `fetchResultForViewer`가 nil (배열 기반)
- 뷰어 복귀 시 정렬/매핑 불일치 방지를 위한 대안:

```swift
protocol GridDataSource: AnyObject {
    // 기존 프로퍼티들...

    /// 뷰어용 PHFetchResult (옵셔널)
    var fetchResultForViewer: PHFetchResult<PHAsset>? { get }

    /// fetchResultForViewer가 nil인 경우 사용할 에셋 ID 배열
    /// 뷰어에서 순서 보장된 네비게이션에 사용
    var orderedAssetIDs: [String] { get }
}

// TrashDataSource 구현

```

### orderedAssetIDs 사용 흐름

`fetchResultForViewer`가 nil인 경우(Trash) 뷰어 진입/복귀 경로:

```swift
// Base의 didSelectItemAt에서 뷰어 진입 시
func openViewer(at assetIndex: Int) {
    let dataSource = gridDataSource

    // 1. fetchResultForViewer가 있으면 기존 방식
    if let fetchResult = dataSource.fetchResultForViewer {
        let coordinator = ViewerCoordinator(fetchResult: fetchResult, ...)
        // ...
    }
    // 2. nil이면 orderedAssetIDs로 동적 PHFetchResult 생성
    else {
        let assetIDs = dataSource.orderedAssetIDs
        let fetchResult = PHAsset.fetchAssets(
            withLocalIdentifiers: assetIDs,
            options: sortedFetchOptions  // creationDate ascending
        )
        let coordinator = ViewerCoordinator(fetchResult: fetchResult, ...)
        // ...
    }
}
```

**정렬 기준 보장:**
- `GridDataSourceDriverAdapter`: GridDataSourceDriver 내부 정렬 따름
- `AlbumDataSource`: PHFetchResult 정렬 따름 (creationDate ascending)
- `TrashDataSource`: loadTrashedAssets()에서 creationDate ascending 정렬

> **검토 필요**: ViewerCoordinator가 PHFetchResult에 강하게 결합되어 있다면,
> 뷰어 쪽 리팩토링도 검토 대상이 될 수 있음

## UICollectionView 책임 경계

### Base가 담당하는 영역

| 항목 | Base 구현 | 비고 |

### prefetchItemsAt padding 가드

```swift
// Base에서 구현
func collectionView(_ collectionView: UICollectionView,
                    prefetchItemsAt indexPaths: [IndexPath]) {
    let padding = paddingCellCount

    // padding 셀 제외하고 실제 에셋만 prefetch
    let assetIDs = indexPaths.compactMap { indexPath -> String? in
        guard indexPath.item >= padding else { return nil }  // padding 가드
        let assetIndex = indexPath.item - padding
        return gridDataSource.assetID(at: assetIndex)
    }

    imagePipeline.preheat(assetIDs: assetIDs, targetSize: thumbnailSize())
}

func collectionView(_ collectionView: UICollectionView,
                    cancelPrefetchingForItemsAt indexPaths: [IndexPath]) {
    let padding = paddingCellCount

    let assetIDs = indexPaths.compactMap { indexPath -> String? in
        guard indexPath.item >= padding else { return nil }  // padding 가드
        let assetIndex = indexPath.item - padding
        return gridDataSource.assetID(at: assetIndex)
    }

    imagePipeline.stopPreheating(assetIDs: assetIDs)
}
```

### paddingCellCount 처리 방안

> **설계 의도**: paddingCellCount는 **상단 패딩**입니다.
> - 앱 UI: 최신 사진이 맨 아래에 있고, 아래에서부터 위로 스크롤
> - 목적: 맨 아래 행이 꽉 차도록 맨 위 행에 빈 셀 추가
> - 예: 7장 사진 + 3열 → paddingCellCount=2 → 상단에 빈 셀 2개

```swift
// Base에서 구현
func collectionView(_ collectionView: UICollectionView,
                    numberOfItemsInSection section: Int) -> Int {
    return gridDataSource.assetCount + paddingCellCount
}

func collectionView(_ collectionView: UICollectionView,
                    cellForItemAt indexPath: IndexPath) -> UICollectionViewCell {
    let cell = collectionView.dequeueReusableCell(...)

    // 빈 셀 처리 (상단 패딩)
    if indexPath.item < paddingCellCount {
        cell.configureAsEmpty()
        return cell
    }

    // 실제 에셋 인덱스 계산
    let assetIndex = indexPath.item - paddingCellCount
    guard let asset = gridDataSource.asset(at: assetIndex) else {
        cell.configureAsEmpty()
        return cell
    }

    // 기본 설정 (이미지 로딩)
    cell.configure(with: asset, thumbnailSize: thumbnailSize())

    // 서브클래스 추가 설정 (템플릿 메서드)
    configureCell(cell, at: indexPath, asset: asset)

    return cell
}
```

## 데이터 갱신 처리 방안

### reloadGridData() 래퍼 메서드

데이터 변경 시 일관된 갱신을 보장하기 위한 래퍼:

```swift
// BaseGridViewController
func reloadGridData() {
    // 1. 컬렉션 뷰 리로드
    collectionView.reloadData()

    // 2. 빈 상태 뷰 갱신
    updateEmptyStateVisibility()

    // 3. 레이아웃 무효화 (paddingCellCount 변경 대응)
    collectionView.collectionViewLayout.invalidateLayout()

    // 4. 선택 모드 UI 갱신 (삭제된 에셋 선택 해제 등)
    if isSelectMode {
        // 삭제된 에셋이 선택 목록에 있으면 제거
        let validAssetIDs = Set((0..<gridDataSource.assetCount).compactMap {
            gridDataSource.assetID(at: $0)
        })
        selectionManager.removeInvalidSelections(validAssetIDs: validAssetIDs)

        // 툴바 갱신
        updateSelectionToolbar(count: selectionManager.selectedCount)
    }
}

func updateEmptyStateVisibility() {
    emptyStateView.isHidden = gridDataSource.assetCount > 0
}
```

### 데이터 변경 이벤트 처리

각 VC는 자체 데이터 변경 이벤트를 수신하고 `reloadGridData()` 호출:

- **GridViewController**: `PHPhotoLibraryChangeObserver` 또는 `GridDataSourceDriver` 이벤트
- **AlbumGridViewController**: `PHPhotoLibraryChangeObserver`
